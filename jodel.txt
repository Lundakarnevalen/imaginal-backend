post: { 	--This is a jodel post, the "head message"
  id: integer,
  votes: integer,
  comments: integer,
  text: string,
  createdAt: Some timestamp,
  voted: boolean,
  color: string (e.g. "#ffffff"),
  isFavourite: boolean
}

comment: {	--This is how a comment for a post looks like
  isComment: true,
  text: String,
  isOp: null or boolean,
  createdAt: Some timestamp
}

reports: [{
  id: boolean,
  votes: integer,
  text: string,
  isComment: true or null
  createdAt: some timestamp
}]

POST /api/jodel/newcomment Body: message<String>, id<Integer>

POST /api/jodel/addfavourite Body: jodelId<Integer>
POST /api/jodel/report/post Body: jodelId<Integer>
POST /api/jodel/report/comment Body: commentId<Integer>, message<String>
POST /api/jodel/delete/report/delete Body: id<Integer>
POST /api/jodel/delete/post Body: jodelId<Integer>
POST /api/jodel/delete/comment Body: commentId<Integer>
